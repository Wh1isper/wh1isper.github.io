下面的内容不一定是最优解，仅供参考。

> 给出两个 非空 的链表用来表示两个非负的整数。其中，它们各自的位数是按照 逆序 的方式存储的，并且它们的每个节点只能存储 一位 数字。
> 
> 如果，我们将这两个数相加起来，则会返回一个新的链表来表示它们的和。
> 
> 您可以假设除了数字 0 之外，这两个数都不会以 0 开头。
> 	
> 	示例：
> 	
> 	输入：(2 -4 -3) + (5 -6 -4)
> 	输出：7 -0 -8
> 	原因：342 + 465 = 807

非常基础的链表操作。

	/**
	 * Definition for singly-linked list.
	 * struct ListNode {
	 *     int val;
	 *     ListNode *next;
	 *     ListNode(int x) : val(x), next(NULL) {}
	 * };
	 */
	class Solution{
	public:
	    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2)
	    {
	        ListNode *temp1 = l1;
	        ListNode *temp2 = l2;
	        ListNode *temp3;
	        ListNode *result = new ListNode(0);
	        ListNode *temp ;
	        bool carry = false;
	        while (temp1 != nullptr || temp2 != nullptr)
	        {
	
	            temp3 = new ListNode(0);
	            temp3->next = nullptr;
	            temp3->val = 0;
	            if (temp1 != nullptr)
	            {
	                temp3->val += temp1->val;
	                temp1 = temp1->next;
	            }
	            if (temp2 != nullptr)
	            {
	                temp3->val += temp2->val;
	                temp2 = temp2->next;
	            }
	            temp3->val = temp3->val + carry;
	            carry = false;
	            if (temp3->val > 9)
	            {
	                temp3->val = temp3->val % 10;
	                carry = true;
	            }
	            temp = result;
	            while (temp->next != nullptr) // 顺序插入时这句话可以略去
	                temp = temp->next;
	            temp->next = temp3;
	        }
	        if(carry)
	        {
	            temp3 = new ListNode(1);
	            temp = result;
	            while (temp->next != nullptr)
	                temp = temp->next;
	            temp->next = temp3;
	        }
	            
	        return result->next;
	    }
	};
